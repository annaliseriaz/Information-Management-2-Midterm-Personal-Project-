    // Auth & Actions
    async function handleRegister() {
        const name = document.getElementById('reg-name').value.trim();
        const email = document.getElementById('reg-email').value.trim().toLowerCase();
        const id_num = document.getElementById('reg-id').value.trim();
        const college = document.getElementById('reg-college').value;
        const course = document.getElementById('reg-course').value;
        const pass = document.getElementById('reg-pass').value;
        const confirm = document.getElementById('reg-confirm').value;

        if (!name || !email || !id_num || !college || !course || !pass || !confirm) {
            showToast('Please fill in all fields.', 'error'); return;
        }
        if (pass !== confirm) { showToast('Passwords do not match.', 'error'); return; }

        showLoading('Creating account...');
        const { data, error } = await db.auth.signUp({
            email, password: pass,
            options: { data: { full_name: name, id_number: id_num, college, course } }
        });

        if (error) { hideLoading(); showToast(error.message, 'error'); return; }
        
        await db.from('user_profiles').insert([{ id: data.user.id, full_name: name, email, id_number: id_num, college, course }]);
        hideLoading();
        showToast('Registration successful!');
        showLogin();
    }

    async function handleLogin() {
        const email = document.getElementById('email-input').value.trim().toLowerCase();
        const pass = document.getElementById('pass-input').value;
        if (!email || !pass) return;

        showLoading('Authenticating...');
        const { data, error } = await db.auth.signInWithPassword({ email, password: pass });
        hideLoading();

        if (error) { showToast('Invalid credentials.', 'error'); return; }

        if (selectedMode === 'admin') {
            const { data: adminRow } = await db.from('admin_users').select('email').eq('email', email).single();
            if (!adminRow) { showToast('Unauthorized access.', 'error'); await db.auth.signOut(); return; }
            enterAdmin();
        } else {
            const { data: profile } = await db.from('user_profiles').select('*').eq('id', data.user.id).single();
            enterRegular(email, profile);
        }
    }

    async function handleGoogleLogin() {
        localStorage.setItem('loginMode', selectedMode);
        await db.auth.signInWithOAuth({ provider: 'google', options: { redirectTo: window.location.href } });
    }

        const { error } = await db.from('visitor_logs').insert([entry]);
        if (error) showToast('Error submitting log.', 'error');
        else {
            showToast('Entry recorded! Have a great day.');
            // Clear form
            ['v-reason', 'visitor-type'].forEach(id => document.getElementById(id).selectedIndex = 0);
        }
    }

    async function logoutUser() { await db.auth.signOut(); location.reload(); }
    async function logoutAdmin() { await db.auth.signOut(); location.reload(); }

    async function loadAdminData() {
        const date = document.getElementById('filter-date').value;
        let query = db.from('visitor_logs').select('*').gte('time_in', `${date}T00:00:00`).lte('time_in', `${date}T23:59:59`);
        
        const { data, error } = await query;
        if (error) return;

        const tbody = document.getElementById('visitor-tbody');
        tbody.innerHTML = data.map(log => `
            <tr>
                <td>${new Date(log.time_in).toLocaleTimeString()}</td>
                <td>${log.full_name}</td>
                <td>${log.id_number}</td>
                <td>${log.college}</td>
                <td>${log.course}</td>
                <td>${log.visitor_type}</td>
                <td>${log.reason}</td>
            </tr>
        `).join('') || '<tr><td colspan="7" style="text-align:center;">No records found.</td></tr>';
        
        document.getElementById('total-v-count').innerText = data.length;
    }

    // Initialize session check
    document.addEventListener('DOMContentLoaded', async () => {
        const { data: { session } } = await db.auth.getSession();
        if (session) {
            const mode = localStorage.getItem('loginMode') || 'regular';
            if (mode === 'admin') enterAdmin();
            else {
                const { data: profile } = await db.from('user_profiles').select('*').eq('id', session.user.id).single();
                enterRegular(session.user.email, profile);
            }
        }
    });
